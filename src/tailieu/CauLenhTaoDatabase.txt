create database librarytesting
go 
use librarytesting
go
create table book(
    bookid int primary key,
	nameBook nvarchar(255),
	description nvarchar(255),
	type nvarchar(255),
	writing nvarchar(255),
	quantity int ,
	yearRelease int
)

create table Users(
    userID int primary key,
	name nvarchar(255),
	phone nvarchar(255),
	address nvarchar(255),
	email nvarchar(255),
	gender nvarchar(255) ,
)

create table transactions(
     transactionId int primary key,
     startDay int,
     endDay int,
     quantity int,
     bookid int FOREIGN KEY REFERENCES book (bookid),
     userID int FOREIGN KEY REFERENCES Users(userID)
)
create table resume(
     bookid int FOREIGN KEY REFERENCES book (bookid),
     userID int FOREIGN KEY REFERENCES Users(userID),
	 transactionId int FOREIGN KEY REFERENCES transactions(transactionId),
	 status nvarchar(255),
	 resumeid int primary key
 )


create table librarian(
    libID int primary key,
	name nvarchar(255),
	phone nvarchar(255),
	address nvarchar(255),
	email nvarchar(255),
	password nvarchar(255),
	gender nvarchar(255) 
 )
